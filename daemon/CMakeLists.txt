# --- CMakeLists.txt ---

# 设定CMake最低版本要求
# Set the minimum required version of CMake
cmake_minimum_required(VERSION 3.15)

# 定义项目名称和语言
# Define the project name and language
project(crfzitd CXX)

# 强制使用C++20标准
# Enforce the C++20 standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# --------------------------------------------------------------------
# 依赖项查找 (Finding Dependencies)
# --------------------------------------------------------------------
# 提示：在此阶段，请确保您已经交叉编译了gRPC和Protobuf，
# 并将SQLite3准备好，让CMake能够通过find_package找到它们。
# You may need to set CMAKE_PREFIX_PATH to point to where you installed them.
# --------------------------------------------------------------------

# 查找gRPC, Protobuf, SQLite3库
# Find gRPC, Protobuf, and SQLite3 libraries
find_package(gRPC CONFIG REQUIRED)
find_package(Protobuf CONFIG REQUIRED)
find_package(SQLite3 REQUIRED)

# --------------------------------------------------------------------
# 源文件与可执行文件定义 (Source Files & Executable Definition)
# --------------------------------------------------------------------

# 添加gRPC和Protobuf生成的源文件 (当前为空，为下一步做准备)
# Add source files generated by gRPC and Protobuf (currently empty, preparing for the next step)
# set(PROTO_SRCS ...)
# set(GRPC_SRCS ...)

# 定义可执行文件crfzitd，并指定其源文件
# Define the executable 'crfzitd' and specify its source files
add_executable(crfzitd
    src/main.cpp
    # ${PROTO_SRCS}
    # ${GRPC_SRCS}
)

# 链接所需的库
# Link the required libraries
target_link_libraries(crfzitd PRIVATE
    gRPC::grpc++          # gRPC核心库
    protobuf::libprotobuf # Protobuf库
    sqlite3               # SQLite3库
)
